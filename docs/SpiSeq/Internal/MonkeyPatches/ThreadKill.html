def kill
  # If there's a magic Thread::Queue thread-local variable, signal it and
  # join; we hope the thread exits quickly after we push to the queue.
  kill_queue = self[:__kill_queue]
  if kill_queue.nil?
    super
  else
    kill_queue << true
    join
  end
end
